CREATE TABLE IF NOT EXISTS public.usuario
(
    "usuarioID" integer NOT NULL DEFAULT nextval('"usuario_usuarioID_seq"'::regclass),
    "nombreUsuario" character varying(30) COLLATE pg_catalog."default" NOT NULL,
    email character varying(60) COLLATE pg_catalog."default" NOT NULL,
    clave character varying(20) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT usuario_pkey PRIMARY KEY ("usuarioID"),
    CONSTRAINT usuario_email_key UNIQUE (email),
    CONSTRAINT "usuario_nombreUsuario_key" UNIQUE ("nombreUsuario")
)
